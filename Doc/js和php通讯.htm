<html>

<head>

<script>

function fileHint(str){

       if(str=="cancel")alert("文件已恢复");

       if(str=="keep")alert("文件已保存");

       //xmlhttp是一个关键对象

           if (window.XMLHttpRequest)

               // IE7+, Firefox, Chrome, Opera, Safari 浏览器执行的代码

               xmlhttp=new XMLHttpRequest();

           else

              //IE6, IE5 浏览器执行的代码

               xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");

       //绑定函数，当页面的状态发生改变的时候调用函数

           xmlhttp.onreadystatechange=function(){

              if (xmlhttp.readyState==4 && xmlhttp.status==200)

                     document.getElementById("fileHint").innerHTML=xmlhttp.responseText;

       }

       xmlhttp.open("GET","fileHint.php?q="+str,true);

       xmlhttp.send();

}

</script>

</head>

<body>

<textarea wrap="off" style="overflow: scroll; width: 500px; height: 250px;" id="fileHint"></textarea><br/>

<form>

       <input type="button" value="取消" onclick="fileHint('cancel')">

       <input type="button" value="保存" onclick="fileHint('keep')">

</form>

</body>

</html>
